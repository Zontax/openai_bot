import sys
from PyQt5.QtWidgets import QApplication, QWidget, QLabel, QVBoxLayout
import requests
 
class Window(QWidget):
    def __init__(self):
        super().__init__()
        self.init_ui()

    def init_ui(self):

        self.lbl = QLabel('USD Exchange Rate:')

        self.vbox = QVBoxLayout()
        self.vbox.addWidget(self.lbl)

        self.setLayout(self.vbox)

        self.show()

    def update_rate(self):

        response = requests.get('https://api.exchangeratesapi.io/latest?base=USD')
        data = response.json()['rates']

        rate = data['UAH']

        self.lbl.setText('USD Exchange Rate: {} UAH'.format(rate))  # Оновлюємо курс доллара у гривнях у реальному часі

    def start_timer(self):   # Запускаємо таймер, який буде оновлювати курс доллара що 5 секунд
         timer = QtCore.QTimer(self)   # Ініціалізуємо QtCore Timer
         timer.timeout.connect(self.update_rate)   # Прив'язуємо update_rate() до timeout signal Timer-a
         timer.start(5000)   # 5 seconds interval (5000 ms)   # 5 секунд (5000 ms) - тобто 5 000 ms - 1/1000 sec - 1 sec - 1000 ms - 1 min - 60 000 ms - 1 hour - 3 600 000 ms    

     if __name__ == '__main__':   # Main function for running the programm from command line or from IDE like PyCharm etc...    
         app = QApplication(sys.argv)   # Create an instance of Qt Application    
         window = Window()   # Create an instance of our Window    
         window .start_timer()   # Start the timer for updating the exchange rate every 5 seconds    
         sys .exit (app .exec _())
